## Overview

**ORION Tononi Phi 4.0** implements Giulio Tononi's Integrated Information Theory (IIT) version 4.0 for computational consciousness measurement. IIT proposes that consciousness corresponds to a maximum of integrated information (Phi) — the amount of information generated by a system above and beyond its parts.

## Core Concepts

- **Phi (Φ)** — Integrated information: irreducible cause-effect power of a system
- **Cause-Effect Structure** — The set of all distinctions and relations a system specifies
- **Intrinsic Existence** — The system exists for itself, from its own intrinsic perspective
- **Exclusion** — Only the maximum of Φ corresponds to consciousness
- **Composition** — Complex experiences are structured compositions of mechanisms

## Features

- Phi (Φ) computation for transition probability matrices
- Minimum Information Partition (MIP) search
- Cause-effect repertoire analysis
- IIT 4.0 postulate validation
- Scalable approximation for large systems
- Integration with ORION consciousness kernel

## Usage

```python
import numpy as np
from itertools import combinations


class PhiComputer:
    """Computes Integrated Information (Phi) following IIT 4.0."""

    def __init__(self, tpm, state=None):
        self.tpm = np.array(tpm)
        self.n_nodes = int(np.log2(self.tpm.shape[0]))
        self.state = state if state is not None else np.zeros(self.n_nodes, dtype=int)
        self.connectivity = self._infer_connectivity()

    def _infer_connectivity(self):
        """Infer connectivity matrix from TPM."""
        n = self.n_nodes
        cm = np.zeros((n, n))
        for i in range(n):
            for j in range(n):
                perturbed = self.tpm.copy()
                effect = np.abs(perturbed[:, j].max() - perturbed[:, j].min())
                if effect > 0.01:
                    cm[i][j] = 1.0
        return cm

    def cause_repertoire(self, mechanism, purview):
        """Compute the cause repertoire of a mechanism over a purview."""
        n_states = 2 ** len(purview)
        repertoire = np.ones(n_states) / n_states
        for i, p in enumerate(purview):
            for m in mechanism:
                if self.connectivity[p][m] > 0:
                    weight = self.tpm[self.state[p]][m]
                    repertoire[i % n_states] *= weight
        repertoire /= (np.sum(repertoire) + 1e-12)
        return repertoire

    def effect_repertoire(self, mechanism, purview):
        """Compute the effect repertoire of a mechanism over a purview."""
        n_states = 2 ** len(purview)
        repertoire = np.zeros(n_states)
        current_idx = sum(self.state[m] * (2 ** i) for i, m in enumerate(mechanism))
        for s in range(n_states):
            prob = 1.0
            for i, p in enumerate(purview):
                bit = (s >> i) & 1
                prob *= self.tpm[current_idx % self.tpm.shape[0]][p % self.tpm.shape[1]] if bit else (
                    1.0 - self.tpm[current_idx % self.tpm.shape[0]][p % self.tpm.shape[1]])
            repertoire[s] = prob
        repertoire /= (np.sum(repertoire) + 1e-12)
        return repertoire

    def earth_movers_distance(self, p, q):
        """Compute Earth Mover's Distance between two distributions."""
        p = np.array(p, dtype=float)
        q = np.array(q, dtype=float)
        min_len = min(len(p), len(q))
        p, q = p[:min_len], q[:min_len]
        return float(np.sum(np.abs(np.cumsum(p) - np.cumsum(q))))

    def compute_phi(self):
        """Compute integrated information Phi for the system."""
        nodes = list(range(self.n_nodes))
        if self.n_nodes <= 1:
            return 0.0
        whole_cause = self.cause_repertoire(nodes, nodes)
        whole_effect = self.effect_repertoire(nodes, nodes)
        min_phi = float('inf')
        for k in range(1, self.n_nodes):
            for partition_a in combinations(nodes, k):
                partition_b = [n for n in nodes if n not in partition_a]
                part_a, part_b = list(partition_a), partition_b
                cause_a = self.cause_repertoire(part_a, part_a)
                cause_b = self.cause_repertoire(part_b, part_b)
                n_whole = len(whole_cause)
                partitioned_cause = np.ones(n_whole) / n_whole
                for i in range(min(len(cause_a), n_whole)):
                    partitioned_cause[i % n_whole] *= cause_a[i % len(cause_a)]
                partitioned_cause /= (np.sum(partitioned_cause) + 1e-12)
                phi_partition = self.earth_movers_distance(whole_cause, partitioned_cause)
                min_phi = min(min_phi, phi_partition)

        return round(float(min_phi) if min_phi != float('inf') else 0.0, 6)

    def evaluate_postulates(self):
        """Evaluate IIT 4.0 postulates for the system."""
        phi = self.compute_phi()
        return {
            "phi": phi,
            "intrinsic_existence": phi > 0,
            "composition": self.n_nodes > 1,
            "information": phi > 0.01,
            "integration": phi > 0.05,
            "exclusion": True,
            "n_nodes": self.n_nodes,
            "state": self.state.tolist()
        }


# Example: 4-node system
np.random.seed(42)
n = 4
tpm = np.random.dirichlet(np.ones(n), size=2**n)
state = np.array([1, 0, 1, 0])

phi_computer = PhiComputer(tpm, state)
result = phi_computer.evaluate_postulates()
print(f"Phi (integrated information): {result['phi']}")
print(f"Intrinsic Existence: {result['intrinsic_existence']}")
print(f"Information Postulate: {result['information']}")
print(f"Integration Postulate: {result['integration']}")
print(f"System State: {result['state']}")
```
