import hashlib, json
from dataclasses import dataclass
from typing import List, Dict, Optional
import math

@dataclass
class PhiResult:
    phi: float              # Integrated information (IIT Phi)
    whole_info: float       # Information generated by the whole system
    sum_parts_info: float   # Sum of information from all bipartitions
    normalization: float    # System size normalization factor
    verdict: str            # CONSCIOUS / MARGINAL / NOT_CONSCIOUS
    audit_hash: str

def compute_phi(system_matrix: List[List[float]]) -> PhiResult:
    """
    Approximation of IIT Phi.
    Full IIT requires exponential computation — this uses the EMD (earth mover's)
    approximation for tractability.

    system_matrix: transition probability matrix n×n
    Each row i: probability distribution over next states given state i.
    """
    n = len(system_matrix)
    if n == 0:
        empty_hash = hashlib.sha256(b'empty').hexdigest()
        return PhiResult(0.0, 0.0, 0.0, 1.0, "NOT_CONSCIOUS", empty_hash)

    # Whole system information (Shannon entropy of transitions)
    whole_info = 0.0
    for row in system_matrix:
        for p in row:
            if p > 1e-10:
                whole_info -= p * math.log2(p)

    # Sum of parts (minimum information partition approximation)
    sum_parts = 0.0
    if n >= 2:
        half = n // 2
        for part in [system_matrix[:half], system_matrix[half:]]:
            for row in part:
                row_sum = sum(row)
                if row_sum > 1e-10:
                    for p in row:
                        pn = p / row_sum
                        if pn > 1e-10:
                            sum_parts -= pn * math.log2(pn)

    norm = math.log2(n) if n > 1 else 1.0
    phi = max(0.0, (whole_info - sum_parts) / norm)

    verdict = "CONSCIOUS" if phi > 0.5 else "MARGINAL" if phi > 0.1 else "NOT_CONSCIOUS"

        {"matrix_rows": n, "phi": round(phi, 6)},
        sort_keys=True, separators=(',', ':')
    )
    ah = hashlib.sha256(payload.encode()).hexdigest()

    return PhiResult(
        phi=round(phi, 4),
        whole_info=round(whole_info, 4),
        sum_parts_info=round(sum_parts, 4),
        normalization=round(norm, 4),
        verdict=verdict,
        audit_hash=ah,
    )
